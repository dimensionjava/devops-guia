
	pipeline {
    agent any
    tools {
        maven 'maven3'
        jdk 'jdk17'
    }

    stages {
        stage('Git checkout') {
            steps {
                git credentialsId: 'github-credentials', url: 'https://github.com/dimensionjava/devops_udemy.git'
            }
        }
        
         stage('Build Application') {
            steps {
                sh "mvn -f billing/pom.xml clean install"
                
            }
        }
        
          stage('Build Docker Image') {
            steps {
               script {  
                   withDockerRegistry(credentialsId: 'dockerhub-credentials', toolName: 'docker') {
                   
                    sh "docker build --build-arg JAR_FILE=/target/*.jar -t billingapp-backend -f billing/Dockerfile billing/"
                   
                   }
               }
                
            }
        }
        
        stage('Tag & Push Docker Image') {
            steps {
               script {  
                   withDockerRegistry(credentialsId: 'dockerhub-credentials', toolName: 'docker') {
                    sh "docker tag billingapp-backend dimensionjava/billingapp-backend:latest"
                    sh "docker push dimensionjava/billingapp-backend:latest"
                   }
               }
                
            }
        }
        
        
     stage('Deploy in Kubernetes') {
      steps {
        withCredentials(bindings: [
            string(credentialsId: 'kubernetes-credentials', variable: 'api_token')
                      ]) {
       sh 'kubectl --token $api_token --server https://192.168.49.2:8443 --insecure-skip-tls-verify=true apply -f billing/hola-deployment.yml '
          }

        }
      }
    }
}
